1 /**
  2  * Model tango_pipe
  3  *
  4  * @type {TangoPipe}
  5  */
  6 TangoPipe = MVC.Model.extend('tango_pipe',
  7     /** @Static */
  8     {
  9 
 10         attributes: {
 11             id: 'string', //host_id/device_id/name
 12             name: 'string',
 13             display_name: 'string'
 14         },
 15         default_attributes: {}
 16     },
 17     /** @Prototype */
 18     {
 19         _get_device_id: function () {
 20             return this.id.substr(0, this.id.lastIndexOf('/'));
 21         },
 22         /**
 23          *
 24          * @param attrs
 25          * @constructor
 26          */
 27         init:function(attrs){
 28             attrs.display_name = attrs.display_name || attrs.name;
 29             this._super(attrs);
 30         },
 31         /**
 32          * @returns {webix.promise}
 33          */
 34         read: function () {
 35             var device = PlatformContext.devices.getItem(this.device_id);
 36 
 37             return device.readPipe(this.name).then(function(resp){
 38                 this.update_attributes(resp);
 39                 return this;
 40             }.bind(this));
 41         },
 42         /**
 43          *
 44          * @param {Object} value
 45          * @returns {webix.promise}
 46          */
 47         write: function (value) {
 48             var device = PlatformContext.devices.getItem(this.device_id);
 49 
 50             return device.writePipe(this.name, value).then(function(resp){
 51                 this.update_attributes(resp);
 52                 return this;
 53             }.bind(this));
 54         },
 55         /**
 56          *
 57          * @returns {'PIPE'}
 58          */
 59         getDataFormat:function(){
 60             return 'PIPE';
 61         }
 62     }
 63 );
