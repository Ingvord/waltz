1 /**
  2  * Model tango_database
  3  *
  4  * @type {TangoDatabase}
  5  */
  6 TangoWebappPlatform.TangoDatabase = MVC.Model.extend('tango_database',
  7     /* @Static */
  8     {
  9 
 10         attributes: {
 11             id: 'string',
 12             info: 'string[]',
 13             device: 'TangoDevice'
 14         },
 15         default_attributes: {}
 16 
 17 
 18     },
 19     /* @Prototype */
 20     {
 21         /**
 22          *
 23          * @param name
 24          * @return {*|Promise}
 25          */
 26         getDeviceInfo: function (name) {
 27 
 28             return this.device.executeCommand("DbGetDeviceInfo", name).then(function (resp) {
 29                 return {
 30                     exported: resp.output.lvalue[0] == 1,
 31                     pid: resp.output.lvalue[1],
 32                     name: resp.output.svalue[0],
 33                     ior: resp.output.svalue[1],
 34                     idl: resp.output.svalue[2],
 35                     admin: resp.output.svalue[3],
 36                     host: resp.output.svalue[4],
 37                     started_at: resp.output.svalue[5], //TODO parse
 38                     stopped_at: resp.output.svalue[6], //TODO parse
 39                     device_class: resp.output.svalue[7]
 40                 };
 41             });
 42         },
 43         /**
 44          * Returns error response if alias is not set - limitation of the native Tango API
 45          *
 46          * @param name
 47          * @return {*|Promise}
 48          */
 49         getDeviceAlias: function (name) {
 50             return this.device.executeCommand("DbGetDeviceAlias", name)
 51                 .then(function(resp){
 52                     return resp.output;
 53                 })
 54         },
 55         /**
 56          *
 57          * @param svalue
 58          */
 59         addDevice: function (svalue) {
 60             return this.device.executeCommand("DbAddDevice", svalue);
 61         },
 62         /**
 63          *
 64          * @param wildcard
 65          */
 66         getDeviceDomainList: function (wildcard) {
 67             return this.device.executeCommand("DbGetDeviceDomainList", wildcard);
 68         },
 69         /**
 70          *
 71          * @param wildcard
 72          */
 73         getDeviceFamilyList: function (wildcard) {
 74             return this.device.executeCommand("DbGetDeviceFamilyList", wildcard);
 75         },
 76         /**
 77          *
 78          * @param wildcard
 79          */
 80         getDeviceMemberList: function (wildcard) {
 81             return this.device.executeCommand("DbGetDeviceMemberList", wildcard);
 82         },
 83         /**
 84          *
 85          * @param {[]} args
 86          * @returns {*|webix.promise}
 87          */
 88         getDeviceProperty: function (args) {
 89             return this.device.executeCommand("DbGetDeviceProperty", args)
 90         },
 91         /**
 92          *
 93          * @param device
 94          * @return {*|webix.promise}
 95          */
 96         deleteDevice: function (device) {
 97             return this.device.executeCommand("DbDeleteDevice", device);
 98         }
 99         //TODO commands
100     }
101 );
