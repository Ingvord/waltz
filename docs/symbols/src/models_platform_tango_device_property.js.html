1 /**
  2  * Model tango_device_property
  3  *
  4  * @type {TangoDeviceProperty}
  5  */
  6 TangoDeviceProperty = MVC.Model.extend('tango_device_property',
  7     /* @Static */
  8     {
  9 
 10         attributes: {
 11             id: 'string',
 12             name: 'string',
 13             device_id: 'string',
 14             values:'[]'
 15         },
 16         default_attributes: {}
 17     },
 18     /* @Prototype */
 19     {
 20         /**
 21          *
 22          * @returns {webix.promise}
 23          */
 24         get:function(){
 25             var device = PlatformContext.devices.getItem(this.device_id);
 26 
 27             return device.fetchProperty(this.name);
 28         },
 29         /**
 30          *
 31          * @param {[]} values
 32          */
 33         set:function(values){
 34             this.values = [];
 35             return this.update(values);
 36         },
 37         /**
 38          *
 39          * @param {[]} values
 40          * @returns {webix.promise}
 41          */
 42         update:function(values){
 43             var device = PlatformContext.devices.getItem(this.device_id);
 44 
 45             var props = Object.create(null);
 46             props[this.name] = this.values.concat(values);
 47             return device.putProperties(props);
 48         },
 49         /**
 50          *
 51          * @returns {'DEVICE_PROPERTY'}
 52          */
 53         getDataFormat: function () {
 54             return 'DEVICE_PROPERTY';
 55         }
 56     }
 57 );
