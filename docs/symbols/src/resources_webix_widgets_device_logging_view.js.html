 12             var top = this;

 14                 rows: [
 15                     {
 16                         id: "logging",
 17                         view: "datatable",

 21                         ]
 22                     },
 23 
 24                     {
 25                         view: "toolbar",

 41                 ]

 45             var top = this.getTopParentView();

 47

 52                 webix.promise.all([
 53                     db.then(function (db) {
 54                         return db.getDeviceProperty([device.name, "logging_level", "logging_target", "logging_rft"]).then(function (resp) {

 56                                 {
 57                                     name: top._labels[resp.output[2]],
 58                                     value: resp.output[4]
 59                                 },

 62                                     value: resp.output[4]
 63                                 },
 64                                 {
 65                                     name: top._labels[resp.output[8]],
 66                                     value: resp.output[10]
 67                                 }
 68                             ];
 69                         })
 70                     }),
 71                     admin.then(function (admin) {
 72                         return admin.getLoggingLevel([top._device.name]);
 73                     }).then(function (resp) {
 74                         return {
 75                             name: top._labels["GetLoggingLevel"],
 76                             value: top._logging_level[resp.output.lvalue[0]]
 77                         };
 78                     }),
 79                     admin.then(function (admin) {
 80                         return admin.getLoggingTarget(top._device.name);
 81                     }).then(function (resp) {
 82                         return {
 83                             name: top._labels["GetLoggingTarget"],
 84                             value: resp.output
 85                         };
 86                     })
 87                 ]).then(function (results) {
 88                     results[0].push(results[1], results[2]);
 89                     return results[0]
 90                 });
 91 
 92 
 93             top.$$('logging').parse(dataPromise);
 94         },
 95         name: "device_logging",
