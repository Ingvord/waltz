 36     /**
 37      * @type {webix.protoUI}
 38      */
 39     var tree = webix.protoUI({
 40         devices_filter: null,
 41         name: 'devices_tree_tree',
 42         /**
 43          *
 44          * @param tango_host
 45          * @param id
 46          * @return {Promise} an array of domains
 47          * @private
 48          */
 49         _expand_tango_host:function(tango_host){
 50             var filter = PlatformContext.UserContext.toDeviceFilter();
 51             var db = tango_host.fetchDatabase();
 52             return webix.promise.all(filter.getDomainFilters().map(function (it) {
 53                 return db
 54                     .fail(function () {
 55                         return [];
 56                     })
 57                     .then(function (db) {
 58                         return db.getDeviceDomainList(it)
 59                     }).then(function (resp) {
 60                         return resp.output.map(function (el) {
 61                             return {value: el, _value: el, webix_kids: true, $css: 'domain'};
 62                         });
 63                     });
 64             })).then(function(filtered_domains){
 65                 return Array.prototype.concat.apply([], filtered_domains); //flatten an array of arrays
 66             });
 67         },
 68 
 69         /**
 70          *
 71          * @param tango_host
 72          * @param domain
 73          * @return {Promise}
 74          * @private
 75          */
 76         _expand_domain:function(tango_host, domain){
 77             var filter = PlatformContext.UserContext.toDeviceFilter();
 78             var db = tango_host.fetchDatabase();
 79             return webix.promise.all(filter.getFamilyFilters(domain).map(function (it) {
 80                 return db
 81                     .then(function (db) {
 82                         return db.getDeviceFamilyList(it);
 83                     }).then(function (resp) {
 84                         return resp.output.map(function (el) {
 85                             return {value: el, _value: el, webix_kids: true, $css: 'family'};
 86                         });
 87                     })
 88             })).then(function(filtered_families){
 89                 return Array.prototype.concat.apply([], filtered_families);//flatten an array of arrays
 90             });
 91         },
 92         /**
 93          *
 94          * @param tango_host
 95          * @param domain
 96          * @param family
 97          * @return {Promise}
 98          * @private
 99          */
100         _expand_family: function(tango_host, domain, family){
101             var filter = PlatformContext.UserContext.toDeviceFilter();
102             var db = tango_host.fetchDatabase();
103             return webix.promise.all(filter.getMemberFilters(domain, family).map(function (it) {
104                 return db.then(function (db) {
105                     return db.getDeviceMemberList(it)
106                         .then(function (resp) {
107                             var count = 0;
108 
109                             if(resp.output.length > 10){
110                                 return resp.output.map(function(member){
111                                     return {
112                                         $css: 'member',
113                                         value: member,
114                                         _value: member
115                                     }
116                                 });
117                             } else {
118                                 return webix.promise.all(resp.output.map(function (member) {
119                                     //TODO extract helper
120                                     var device_name = [domain, family, member].join("/");
121 
122                                     return db.getDeviceAlias(device_name)
123                                         .then(function (alias) {
124                                             return {
125                                                 $css: 'member',
126                                                 value: alias,
127                                                 _value: member
128                                             }
129                                         })
130                                         .fail(function () {
131                                             return {
132                                                 $css: 'member',
133                                                 value: member,
134                                                 _value: member
135                                             }
136                                         });
137                                 }));
138                             }
139                         });
140                 })
141             })).then(function(filtered_members){
142                 return Array.prototype.concat.apply([], filtered_members);//flatten an array of arrays
143             });
144         },
145         /**
146          *
147          * @param context
148          * @private
149          */
150         _get_data: function (context) {
151             var data = {
152                 id: 'root',
153                 value: context.rest.url,
154                 open: true,
155                 $css: 'rest_host',
156                 data: []
157             };
158 
159             TangoWebappHelpers.iterate(context.tango_hosts, function (it) {
160                 data.data.push({
161                     id: it.id,
162                     value: it.id,
163                     //TODO bind host node data to context.devices
164                     webix_kids: it.is_alive,
165                     $css: 'tango_host'
166                 });
167             });
168 
169             return data;
170         },
171         /**
172          *
173          * @param item
174          * @private
175          */
176         _get_host_id: function (item) {
177             switch (item.$level) {
178                 case 5://member
179                     (item = this.getItem(item.$parent));
180                 case 4://family
181                     (item = this.getItem(item.$parent));
182                 case 3://domain
183                     return this.getItem(item.$parent).id;
184                 case 2://tango_host
185                     return item.id;
186                 default:
187                     TangoWebappPlatform.debug("_get_host_id called for wrong item: " + item.id);
188             }
189         },
190         /**
191          *
192          * @param item
193          * @private
194          */
195         _get_device_name: function (item) {
196             var member = item._value;
197             var family = this.getItem(item.$parent)._value;
198             var domain = this.getItem(this.getItem(item.$parent).$parent)._value;
199             return [domain, family, member].join('/');
200         },
201         /**
202          *
203          * @param {PlatformContext} context [optional - will use global PlatformContext instead]
204          */
205         updateRoot: function (context) {
206             var context = context || PlatformContext;
207             this.clearAll();
208 
209             var data = this._get_data(context);
210 
211             this.parse([data]);
212         },
213         $init: function (config) {
214             var context = config.context;
215 
216             var data = this._get_data(context);
217             webix.extend(config, {
218                 data: [data]
219             });
220 
221             webix.ui(tree_context_menu).attachTo(this);
222         },
223         defaults: {
224             type: 'lineTree',
225             select: true,
226             on: {
227                 onBeforeContextMenu: function (id, e, node) {
228                     var item = this.getItem(id);
229                     if(item.$level === 5){
230                         this.$$("devices_tree_context_menu").config.master = this;
231                         this.select(id);
232                         return true;
233                     }
234                     return false;
235                 },
236                 onItemDblClick:function(id){
237                     var item = this.getItem(id);
238                     if (!item) return false;//TODO or true
239 
240                     var tango_host_id;
241                     var device_name;
242                     switch (item.$level) {
243                         case 5://member
244                             tango_host_id = this._get_host_id(item);
245                             device_name = this._get_device_name(item);
246 
247                             PlatformApi.PlatformUIController().expandDeviceTree();
248                             break;
249                         default:
250                             TangoWebappHelpers.debug("device_tree#onItemDblClick " + id);
251                     }
252                 },
253                 /**
254                  * Happens before click events
255                  *
256                  * @param id
257                  * @return {boolean}
258                  */
259                 onAfterSelect: function(id){
260                     var item = this.getItem(id);
261                     if (!item) return false;//TODO or true
262 
263                     var tango_host_id;
264                     var tango_host;
265                     var device_name;
266                     switch (item.$level) {
267                         case 2://tango host
268                         case 3://domain
269                         case 4://family
270                             tango_host_id = this._get_host_id(item);
271                             PlatformContext.tango_hosts.setCursor(tango_host_id);
272                             break;
273                         case 5://member
274                             tango_host_id = this._get_host_id(item);
275                             device_name = this._get_device_name(item);
276 
277                             PlatformContext.tango_hosts.setCursor(tango_host_id);
278 
279                             tango_host = PlatformContext.tango_hosts.getItem(tango_host_id);
280                             tango_host.fetchDevice(device_name).then(function (device) {
281                                 PlatformContext.devices.setCursor(device.id);
282                             });
283                             break;
284                         default:
285                             TangoWebappHelpers.debug("device_tree#clickOnItem " + id);
286                     }
287                 },
288                 onDataRequest: function (id, cbk, url) {
289                     var item = this.getItem(id);
290 
291                     var tango_host_id = this._get_host_id(item);
292                     var tango_host = PlatformContext.tango_hosts.getItem(tango_host_id);
293 
294                     var self = this;
295                     switch (item.$level) {
296                         case 2://tango_host
297                             this._expand_tango_host(tango_host).then(function(domains){
298                                 self.parse({
299                                     parent: id,
300                                     data: domains
301                                 });
302                             });
303                             return false;
304                         case 3://domain
305                             this._expand_domain(tango_host, item._value).then(function(families){
306                                 self.parse({
307                                     parent: id,
308                                     data: families
309                                 })
310                             });
311                             return false;
312                         case 4://family
313                             this._expand_family(tango_host,this.getItem(item.$parent)._value,item._value).then(function(members){
314                                 self.parse({
315                                     parent: id,
316                                     data: members
317                                 });
318 
319                                 //load device aliases sequentially if there are too many members
320                                 if(members.length > 10 ){
321                                     members.forEach(function(member){
322                                         tango_host.fetchDatabase().then(function(db){
323                                             var device_name = [self.getItem(item.$parent)._value,item._value, member._value].join("/");
324 
325                                             return db.getDeviceAlias(device_name)
326 
327                                         }).then(function (alias) {
328                                             member.value = alias;
329                                             self.parse({
330                                                 parent: id,
331                                                 data: [
332                                                     member
333                                                 ]
334                                             })
335                                         })
336                                     })
337                                 }
338                             }, function (err) {
339                                 debugger
340                             });
341                     }
342 
343                     return false;//block further execution
344                 },
345                 "user_context_controller.found subscribe": function (event) {
346                     var user_context = event.data;
347                     event.controller.devices_filter = new DeviceFilter({
348                         user: user_context.user,
349                         value: user_context.device_filters
350                     });
351                 },
352                 "user_context_controller.add_tango_host subscribe": function (event) {
353                     event.controller.parse({
354                         parent: 'root',
355                         data: [{
356                             id: event.data,
357                             value: event.data,
358                             $css: 'tango_host',
359                             webix_kids: true
360                         }
361                         ]
362                     });
363                 },
364                 "user_context_controller.delete_tango_host subscribe": function (event) {
365                     event.controller.remove(event.data);
366                 },
367                 "platform_api.ui.initialized subscribe": function (event) {
368                     event.controller.updateRoot(event.data.context);
369                 },
370                 "platform_context.set_rest subscribe": function (event) {
371                     event.controller.updateRoot(event.data);
372                 }
373             }
374         }
