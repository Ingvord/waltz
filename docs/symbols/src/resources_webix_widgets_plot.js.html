 14     var image_plot = webix.protoUI({
 15         value: "",
 16         name: "image",
 17         /**
 18          *
 19          * @param {data:[], width:int, height:int} value
 20          */
 21         update: function (resp) {
 22             var data = [];
 23             for (var i = 0, j = 0; i < resp.value.height; ++i, j += resp.value.width) {
 24                 data.push(resp.value.data.slice(j, j + resp.value.width))
 25             }
 26 
 27             var ratio = resp.value.width / resp.value.height;
 28             var height = Math.max(this.$height, Math.min(this.$height, resp.value.height));
 29             var layout = {
 30                 title: "Data acquired @ " + new Date(resp.timestamp),
 31                 autosize: false,
 32                 width: height * ratio,//assume square
 33                 height: height,
 34                 margin: margins
 35             };
 36 
 37             Plotly.relayout(this.getNode(), layout);
 38             Plotly.restyle(this.getNode(), 'z', [data]);
 39         },
 40         $init: function (config) {
 41             this.$ready.push(function () {
 42                 Plotly.newPlot(this.getNode(), [{
 43                     z: [],
 44                     type: 'heatmap'
 45                 }]);
 46             });
 47         }
 48     }, webix.IdSpace, webix.ui.view);

 56     /**
 57      * @type {webix.protoUI}
 58      */
 59     var scalar_plot = webix.protoUI(
 60         {
 61             name: 'scalar',
 62             _traces: null,
 63             _relayout: function (layout) {
 64                 Plotly.relayout(this.getNode(), layout);
 65             },
 66             /**
 67              *
 68              * @param trace name
 69              * @param {[]} x timestamps
 70              * @param {[]} y values
 71              * @param {int} ndx
 72              */
 73             addTrace: function (trace, x, y, ndx) {
 74                 Plotly.addTraces(this.getNode(), {
 75                     x: x.map(function (time) {
 76                         return new Date(time);
 77                     }),
 78                     y: y,
 79                     name: trace
 80                 }, ndx);
 81                 this._traces[ndx] = trace;
 82                 this._relayout({
 83                     autosize: false,
 84                     width: this.$width,
 85                     height: this.$height,
 86                     margin: margins
 87                 });
 88             },
 89             /**
 90              *
 91              * @param {int} ndx same as used in addTrace
 92              */
 93             deleteTrace: function (ndx) {
 94                 Plotly.deleteTraces(this.getNode(), [ndx]);
 95                 this._traces.splice(ndx,1);
 96                 this._relayout({
 97                     autosize: false,
 98                     width: this.$width,
 99                     height: this.$height,
100                     margin: margins
101                 });
102             },
103             /**
104              *
105              * @param {[]} traces an array of traces indices
106              * @param {[]} times
107              * @param {[]} data an array of data arrays
108              */
109             updateTraces: function (traces, times, data) {
110                 Plotly.extendTraces(this.getNode(), {
111                     x: times.map(function (time) {
112                         return [new Date(time)];
113                     }),
114                     y: data.map(function (data) {
115                         return [data];
116                     })
117                 }, traces);
118                 //TODO check if required
119                 this._relayout({
120                     autosize: false,
121                     width: this.$width,
122                     height: this.$height,
123                     margin: margins
124                 });
125             },
126             /**
127              * Updates this plot with single data item
128              *
129              * @param {{timestamp: int, value: data}} data
130              */
131             update: function (data) {
132                 this.updateMulti([data]);
133             },
134             /**
135              * Updates this plot with multiple data items
136              *
137              * @param {[{timestamp: int, value: data}]} data
138              */
139             updateMulti:function(data){
140                 Plotly.extendTraces(this.getNode(), {
141                     x: [data.map(function(el){ return new Date(el.timestamp)})],
142                     y: [data.map(function(el){ return el.value;})]
143                 }, [0]);
144                 //TODO check if required
145                 this._relayout({
146                     title: "Data acquired @ " + new Date(data.pop().timestamp),
147                     autosize: false,
148                     width: this.$width,
149                     height: this.$height,
150                     margin: margins
151                 });
152             },
153             clear:function(){
154                 Plotly.deleteTraces(this.getNode(), this._traces.map(function(el, ndx){ return ndx;}));
155                 this._traces.forEach(function(trace){
156                     Plotly.addTraces(this.getNode(), {
157                         x: [],
158                         y: [],
159                         name : trace
160                     });
161                 }.bind(this));
162                 Plotly.relayout(this.getNode(),
163                     {
164                         title: ""
165                     });
166             },
167             _newPlot: function (config) {
168                 this._traces = [];
169                 Plotly.newPlot(this.getNode(), [], {
170                     width: this.$width,
171                     height: this.$height,
172                     showlegend: true
173                 });
174             },
175             $init: function (config) {
176                 // webix.extend(config, this._ui(config));
177                 this.$ready.push(this._newPlot.bind(this, config));
178                 this.$ready.push(function () {
179                     if (!config.empty) {
180                         this.addTrace(config.name, [], [], 0);
181                         this.update(config);
182                     }
183                 }.bind(this));
184                 this.$ready.push(function () {
185                     var self = this;
186                     var node = this.getNode();
187                     webix.ui({
188                         view: "contextmenu",
189                         data: [
190                             "Clear"
191                         ],
192                         master: node, //  ID of a DIV container
193                         on: {
194                             onItemClick: function (id) {
195                                 self.clear();
196                             }
197                         }
198                     });
199                 }.bind(this));
200             }
201         }, webix.IdSpace, webix.ui.view);
