 58 TangoWebappPlatform.TangoRestApiRequest = MVC.Model.extend('tango_rest_api_request',
 59     /* @Static */
 60     {
 61         _id: 1,
 62         attributes: {
 63             id: 'number',
 64             url: 'string',
 65             type: 'string',
 66             result: 'object',
 67             failure: 'object'
 68         },
 69         default_attributes: {
 70             result: null,
 71             failure: null
 72         },
 73         init:function(){
 74             this._super();
 75             //do not store requests in production
 76             if(MVC.env() === 'production') {
 77                 this.store_type = TangoWebappPlatform.DummyStore;
 78                 this.store = new TangoWebappPlatform.DummyStore();
 79             }
 80         }
 81     },
 82     /* @Prototype */
 83     {
 84         //promise factory
 85         promise: null,
 86         transport: null,
 87         init: function (params) {
 88             this._super(MVC.Object.extend(params, {id: this.Class._id++}));
 89             this.transport = webix.ajax;
 90             this.promise = webix.promise;
 91         },
 92         /**
 93          *
 94          * @param {} resp
 95          */
 96         add_errors:function(resp){
 97             resp.errors = resp.errors.map(function(error){ return TangoWebappHelpers.newTangoError(error)});
 98             this._super(resp.errors);
 99         },
100         /**
101          *
102          * @param resp
103          * @returns {*}
104          * @private
105          */
106         _success: function (resp) {
107             var json = {};
108             if (resp.text().length > 0) {
109                 json = resp.json();
110 
111                 if (json.quality === 'FAILURE') {
112                     this.add_errors(json);
113                     this.failure = json;
114                     OpenAjax.hub.publish("tango_webapp.rest_failure", {data: this});
115                     throw json;
116                 }
117             }
118             this.result = json;
119             OpenAjax.hub.publish("tango_webapp.rest_success", {data: this});
120             return json;
121         },
122 
123         /**
124          *
125          * @param resp
126          * @private
127          */
128         _failure: function (resp) {
129             var json;
130             try {
131                 json = JSON.parse(resp.responseText);
132             } catch (e) {
133                 json = {
134                     errors: [
135                         {
136                             reason: resp.status,
137                             description: resp.responseText ? resp.responseText : "Unspecified error",
138                             severity: 'ERR',
139                             origin: this.url
140                         }
141                     ],
142                     quality: 'FAILURE',
143                     timestamp: +new Date()
144                 }
145             }
146             this.add_errors(json);
147             this.failure = json;
148             OpenAjax.hub.publish("tango_webapp.rest_failure", {data: this});
149             throw json;
150             //TODO move to platform.ui.controller
151             // if (resp.errors && resp.errors.length > 0) //tango rest specific
152             //     for (var i = 0, size = resp.errors.length; i < size; ++i) {
153             //         TangoWebapp.error(resp.errors[i].reason + ": " + resp.errors[i].description);
154             //     }
155             // else { //general failure
156             //     console.error("Unexpected error");
157             //     throw resp;
158             // }
159         },
160         /**
161          *
162          * @returns {TangoRestApiRequest}
163          */
164         hosts: function (host) {
165             this.url += '/hosts/';
166             this.url += host;

221         /**
222          *
223          * @event {OpenAjax} tango_webapp.rest_success
224          * @event {OpenAjax} tango_webapp.rest_failure
225          * @returns {webix.promise}
226          */
227         exec: function (argin) {
228             return this.put("", argin);
229         },
230 
231 
232         /**
233          *
234          * @event {OpenAjax} tango_webapp.rest_success
235          * @event {OpenAjax} tango_webapp.rest_failure
236          * @returns {webix.promise}
237          */
238         get: function (what) {
239             if (this.result != null) return this.promise.resolve(this.result);
240             if (this.failure != null) return this.promise.reject(this.failure);
241             if (what) this.url += what;
242             this.type = "GET";
243             OpenAjax.hub.publish("tango_webapp.rest_send", {data: this});
244             return this.transport().get(this.url).then(this._success.bind(this)).fail(this._failure.bind(this));
245         },
246 
247         /**
248          *
249          * @event {OpenAjax} tango_webapp.rest_success
250          * @event {OpenAjax} tango_webapp.rest_failure
251          * @returns {webix.promise}
252          */
253         put: function (what, data) {
254             if (this.result != null) return this.promise.resolve(this.result);
255             if (this.failure != null) return this.promise.reject(this.failure);
256             if (what) this.url += what;//TODO if no what is provided data will be treated as what -> failure
257             this.type = "PUT";
258             OpenAjax.hub.publish("tango_webapp.rest_send", {data: this});
259             return this.transport().headers({
260                 "Content-type": "application/json"
261             }).put(this.url, (typeof data == 'object') ? JSON.stringify(data) : data).then(this._success.bind(this)).fail(this._failure.bind(this));
262         },
263 
264         /**
265          *
266          * @event {OpenAjax} tango_webapp.rest_success
267          * @event {OpenAjax} tango_webapp.rest_failure
268          * @returns {webix.promise}
269          */
270         "delete": function (what) {
271             if (this.result != null) return this.promise.resolve(this.result);
272             if (this.failure != null) return this.promise.reject(this.failure);
273             if (what) this.url += what;
274             this.type = "DELETE";
275             OpenAjax.hub.publish("tango_webapp.rest_send", {data: this});
276             return this.transport().del(this.url).then(this._success.bind(this)).fail(this._failure.bind(this));
277         }
