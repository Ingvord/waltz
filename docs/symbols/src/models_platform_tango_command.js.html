1 /**
  2  * Model tango_command
  3  *
  4  * @type {TangoCommand}
  5  */
  6 TangoCommand = MVC.Model.extend('tango_command',
  7     /** @Static */
  8     {
  9 
 10         attributes: {
 11             id: 'string',//host_id/device_id/name
 12             device_id: 'string',
 13             name: 'string',
 14             display_name: 'string',
 15             info: 'object',
 16             input: 'any',
 17             output: 'any'
 18             //TODO history
 19         },
 20         default_attributes: {}
 21     },
 22     /** @Prototype */
 23     {
 24         /**
 25          *
 26          * @param attrs
 27          * @constructor
 28          */
 29         init: function (attrs) {
 30             attrs.display_name = attrs.display_name || attrs.name;
 31             if(attrs.name === 'State' || attrs.name === 'Status') attrs.id = attrs.id + '_cmd'; //TODO may append _cmd multiple times
 32 
 33             this._super(attrs);
 34         },
 35         /**
 36          *
 37          * @param argin
 38          *
 39          * @returns {webix.promise}
 40          */
 41         execute: function (argin) {
 42             var device = PlatformContext.devices.getItem(this.device_id);
 43 
 44             this.update_attributes({
 45                 input: argin
 46             });
 47             return device.executeCommand(this.name, argin).then(function(resp){
 48                 this.update_attributes(resp);
 49                 return this;
 50             }.bind(this));
 51         },
 52         /**
 53          *
 54          * @returns {'COMMAND'}
 55          */
 56         getDataFormat:function(){
 57             return 'COMMAND';
 58         }
 59     }
 60 );
