1 /**
  2  * Model application_context
  3  *
  4  * @type {PlatformContext}
  5  */
  6 TangoWebappPlatform.PlatformContext = MVC.Model.extend('platform_context',
  7     /* @Static */
  8     {
  9         attributes: {
 10             id: 'string',
 11             UserContext: '{}'
 12         },
 13         default_attributes: {
 14             id: 'singleton'
 15         },
 16         /**
 17          *
 18          * @param {Object} attrs - must have rest:TangoRestApi and user_context:UserContext
 19          */
 20         create: function (attrs) {
 21             //TODO make this return value
 22             PlatformContext = this.create_as_existing(attrs);
 23 
 24             var user_context = PlatformContext.UserContext;
 25 
 26             var tango_hosts = [];
 27             for (var tango_host in user_context.tango_hosts) {
 28                 if (!user_context.tango_hosts.hasOwnProperty(tango_host)) continue;
 29 
 30                 tango_hosts.push(tango_host);
 31             }
 32 
 33 
 34             var rest = PlatformContext.rest;
 35 
 36 
 37             rest.promise.all(
 38                 tango_hosts.map(function (it) {
 39                     return rest.fetchHost(it)
 40                         .fail(function (failed_host) {
 41                             TangoWebappHelpers.error("Failed to load Tango host " + failed_host.id);
 42                             return failed_host;//prevent promise.all from failing
 43                         })
 44                 })
 45             ).then(PlatformContext.tango_hosts.parse.bind(PlatformContext.tango_hosts)
 46             ).then(this.publish.bind(this, "create", {data: PlatformContext})
 47             ).fail(function(err){
 48                 TangoWebappHelpers.error("Failed to create PlatformContext! See console...", err);
 49                 debugger
 50             });
 51         }
 52     },
 53     /* @Prototype */
 54     {
 55         //TODO association
 56         UserContext: null,
 57         rest: null,
 58         tango_hosts: null,
 59         devices: null,
 60         /**
 61          *
 62          * @param {UserContext} v
 63          *
 64          * @event {OpenAjax} platform_context.set_user_context
 65          */
 66         set_UserContext: function (v) {
 67             this.UserContext = v;
 68             this.publish("set_user_context", {data: this});
 69         },
 70         /**
 71          *
 72          * @param {TangoRestApi} v - new rest api
 73          *
 74          * @event {OpenAjax} platform_context.set_rest
 75          */
 76         set_rest: function (v) {
 77             this.rest = v;
 78             this.UserContext.update_attributes({
 79                 rest_url: v.url
 80             });
 81 
 82             TangoWebappHelpers.iterate(this.tango_hosts, function (it) {
 83                 it.rest = v;
 84             });
 85 
 86             this.publish("set_rest", {data: this});
 87         },
 88         //TODO jmvc attributes clashes with tango attributes
 89         //attributes: null,
 90         /**
 91          *
 92          * @param attrs
 93          * @constructor
 94          */
 95         init: function (attrs) {
 96             this.tango_hosts = new webix.DataCollection({
 97                 defaultData: TangoHost.default_attributes
 98             });
 99             this.devices = new webix.DataCollection({
100                 defaultData: webix.extend(TangoDevice.default_attributes, {
101                     attrs: new webix.DataCollection(),
102                     commands: new webix.DataCollection(),
103                     pipes: new webix.DataCollection()
104                 })
105             });
106 
107             this._super(attrs);//calls set_rest
108         },
109         /**
110          * Destroys this instance
111          *
112          * @event {OpenAjax} platform_context.destroy
113          */
114         destroy: function () {
115             this._super();
116             this.UserContext.save();
117             //TODO clear all data
118             PlatformContext = null;
119         }
120     }
121 );
